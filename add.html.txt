<script>
requireLogin();
let user = getUser();

function post() {
  let listings = getData("listings");
  listings.push({
    title: title.value,
    description: description.value,
    price: price.value,
    seller: user.username
  });
  setData("listings", listings);
  alert("Listing posted");
  window.location.href = "index.html";
}
